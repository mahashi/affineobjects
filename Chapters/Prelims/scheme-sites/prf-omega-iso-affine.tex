\begin{lemma}\label[lemma]{862}
Let $R$ be a ring.
Let $M$ be a $R$-module.
$\omega^2: \sect{\Spec{R}}{\pstilde{M}} \rightarrow \sect{\Spec{R}}{\stilde{M}}$ is an isomorphism.
\end{lemma}

\begin{proof}
Let $\Ip \subset R$ be a prime ideal.
As stated in \Cref{75a}, we may use results from \cite[Tag 01BH]{stacks} in this setting.
We will use that $\stilde{M}_{\Ip} = M \tensor_R R_{\Ip}$.
Localized at every prime ideal of $R$
the map $\omega^2$ sends $m$ to $m\tensor 1$,
hence is the inverse of the multiplication map which is an isomorphism.
Hence non-localized $\omega^2$ is an isomorphism.
\end{proof}

\begin{corollary}\label[corollary]{872}
Let $X$ be a scheme.
Let $\Spec{R}$ be an open subset of $X$.
Let $M$ be a $\sect{X}{O}$-module.
Then $\omega^2: \sect{\Spec{R}}{\pstilde{M}} \rightarrow \sect{\Spec{R}}{\stilde{M}}$ is an isomorphism. 
\end{corollary}
