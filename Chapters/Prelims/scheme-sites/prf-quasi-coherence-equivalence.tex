\begin{lemma}\label[lemma]{873}
For a sheaf of modules $\sheaf{F}$ on an affine scheme $\Spec{R}$, 
the following are equivalent
\begin{itemize}
	\item $\sheaf{F}$ is quasi-coherent
	\item $\sheaf{F}$ is of the form $\stilde{M}$ for $M=\sect{\Spec{R}}{F}$.
\end{itemize}
\end{lemma}
