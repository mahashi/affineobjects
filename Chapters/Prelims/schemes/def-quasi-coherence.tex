\begin{definition}[Tilde definition]\label[definition]{Tilde definition}
Let $\sheaf{F}$ be a sheaf of modules on scheme $X$.
Then $\sheaf{F}$ is quasi-coherent \iff 
for any $\Spec{R} \subset X$ it is of the form $\stilde{M}$ for some $R$-module after restricting.
\end{definition}

\begin{lemma}
Let $\sheaf{F}$ be a sheaf of modules on scheme $X$.
Then $\sheaf{F}$ is quasi-coherent \iff 
for any $D(f) = \Spec{S_f} \rightarrow \Spec{S}$ opens in $X$ with $f\in S$ 
the induced map
\[ \sect{\Spec{S}}{F} \tensor_{S} S_f \rightarrow \sect{D(f)}{F}\]
is an isomorphism.
\end{lemma}
