\begin{definition}[Tilde definition]\label[definition]{Tilde definition}
Let $\sheaf{F}$ be a sheaf of modules on scheme $X$.
Then $\sheaf{F}$ is quasi-coherent \iff 
for any $\Spec{R} \subset X$ it is of the form $\stilde{M}$ for some $R$-module after restricting.
\end{definition}

\begin{definition}[Restrictive definition]\label[definition]{Restrictive definition}
Let $\sheaf{F}$ be a sheaf of modules on scheme $X$.
Then $\sheaf{F}$ is quasi-coherent \iff 
for any $\Spec{R} \rightarrow \Spec{S}$ the induced map
\[ \sect{\Spec{S}}{F} \tensor_{\sect{X}{O}} \sect{\Spec{R}}{O} \rightarrow \sect{\Spec{R}}{F}\]
is an isomorphism.
\end{definition}
