\begin{lemma}\label[lemma]{a38}
Let $f:\cat{C}\rightarrow \cat{D}$ be a morphism of ringed sites.
Let $\sheaf{F}$ be a quasi-coherent sheaf of modules on $\cat{D}$.
Then $\direct{f}\sheaf{F}$ is quasi-coherent.
\end{lemma}

\begin{proof}
One can view $f$ as a morphism of ringed topoi $(\direct{f},\inverseimg{f}, f^{\#})$.
See \cite[\href{http://stacks.math.colombia.edu/tag/082T}{Tag 082T}]{stacks}.
\end{proof}
