\begin{definition}[Support]\label[definition]{50d}
Let $\rspace{X}{O}$ be a locally ringed space.
The support of a global section is
\[D_X(a) = \{x\in X \st a\not\in \ker(x)\}.\]
\end{definition}

\begin{definition}[Support of module]\label[definition]{513}
Let $\rspace{X}{O}$ be a locally ringed space.
Let $\module{M}$ be a $\sheaf{O}$-module on this space.
The support $\Sup(\module{M})$ of this module is the subspace
\[\set{x \in X \st \module{M}_x = 0} \subset \rspace{X}{O}.\]
\end{definition}

\begin{definition}[Set cut out by ideal]\label[definition]{51a}
Let $\rspace{X}{O}$ be a scheme.
Define the set cut out by an ideal $I$ of $\sect{X}{O}$ to be
\[V_X(I) = \Sup(\Lambda_X(\frac{\sheaf{O}(X)}{I})).\]
\end{definition}
