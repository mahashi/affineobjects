\begin{lemma}\label[lemma]{152}
Let $\rsite{C}{T}{O}$ be a ringed site.
Let $a \in \cat{C}$.
Let $\cover{b_i}{a}$ be a caffine covering on $a$.
Assume every map $b_i \xrightarrow{i} a$ is restrictive.
Then the counit $\counit$ of the adjunction $\mainadjunction{a}$
is a natural isomorphism.
\end{lemma}

\begin{proof}
Let $\module{F}$ be a quasi-coherent sheaf module. 
Set $M = \sections{a}{F}$.
Set $M_i = \sections{b_i}{F}$.
Set $\beta_i = \direct{i}\counit_{\module{F},a}$.
By \Cref{702}, we have$\beta_i \iso \counit_{\direct{i}\module{F},b_i}$.
Since $b_i$ is caffine $\beta_i$ is an isomorphism.
Hence $\counit$ is an isomorphism between sheafs, since it is locally an isomorphism.
\end{proof}
