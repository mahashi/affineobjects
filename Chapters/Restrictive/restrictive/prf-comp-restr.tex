\begin{lemma}\label[lemma]{cf6}
Let $\rsite{C}{T}{O}$ be a ringed site.
Let $f:b\rightarrow a \in \cat{C}$ and $g:c\rightarrow b \in \cat{C}$ be morphisms.
\begin{enumerate}
	\item If $fg$ and $f$ are restrictive, then $g$ is.
	\item if $f$ and $g$ are restrictive, then $fg$. 
\end{enumerate}

If $\sect{b}{O} \rightarrow \sect{c}{O}$ is faithfully flat
then $fg$ and $g$ restrictive implies $f$ is restrictive.
\end{lemma}

\begin{proof}

Consider the following diagram
\begin{center}
	\begin{tikzcd}[row sep = large, column sep = large]
		\sect{a}{G} \tensor_{\sect{a}{O}} \sect{c}{O} \ar{r}{\widehat{f} \tensor_{\sect{b}{O}} \sect{c}{O}} \ar[equal]{d}
		& \sect{b}{G} \tensor_{\sect{b}{O}} \sect{c}{O} \ar{d}{\widehat{g}} \\
		\sect{a}{G} \tensor_{\sect{a}{O}} \sect{c}{O} \ar{r}{\widehat{fg}} 
		& \sect{c}{G},\\
	\end{tikzcd}
\end{center}

where $\widehat{\blank}$ is as in lemma ?.

This diagram commutes: going either direction sends $g\tensor r$ to $rg$.
The results follows from commutativity.
\end{proof}
